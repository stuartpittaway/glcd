00103 /*
00104  *  _ d e l a y _ l o o p _ 3( uint32_t __count )
00105  *
00106  * This delay loop is not used in the code below: It is
00107  * a supplement to the _delay_loop_1() and _delay_loop_2()
00108  * within standard WinAVR <arv/delay.h> giving a wider
00109  * (32 bit) delay range.
00110  *
00111  */
00112 static __inline__ void
00113 _delay_loop_3( uint32_t __count )
00114 {
00115     __asm__ volatile (
00116         "1: sbiw %A0,1" "\n\t"
00117         "sbc %C0,__zero_reg__" "\n\t"
00118         "sbc %D0,__zero_reg__" "\n\t"
00119         "brne 1b"
00120         : "=w" (__count)
00121         : "0" (__count)
00122     );
00123 }

00126 /*
00127  *  _ d e l a y _ l o o p _ 1 _ x( uint8_t __count )
00128  *  _ d e l a y _ l o o p _ 2 _ x( uint16_t  __count )
00129  *  _ d e l a y _ l o o p _ 4 _ x( uint32_t __count )
00130  *
00131  *  These delay loops always have exactly 4(8) cycles per loop.
00132  *  They use a 8/16/32 bit register counter respectively.
00133  *
00134  */
00135 static __inline__ void      /* exactly 4 cycles/loop, max 2**8 loops */
00136 _delay_loop_1_x( uint8_t __n )
00137 {                                               /* cycles per loop      */
00138     __asm__ volatile (                          /* __n..one        zero */
00139         "1: dec  %0"   "\n\t"                   /*    1             1   */
00140         "   breq 2f"   "\n\t"                   /*    1             2   */
00141         "2: brne 1b"   "\n\t"                   /*    2             1   */
00142         : "=r" (__n)                            /*  -----         ----- */
00143         : "0" (__n)                             /*    4             4   */
00144     );
00145 }

00175 /*
00176  *
00177  *  _ d e l a y _ c y c l e s (double __ticks_d)
00178  *
00179  *  Perform an accurate delay of a given number of processor cycles.
00180  *
00181  *  All the floating point arithmetic will be handled by the
00182  *  GCC Preprocessor and no floating point code will be generated.
00183  *  Allthough the parameter __ticks_d is of type 'double' this
00184  *  function can be called with any constant integer value, too.
00185  *  GCC will handle the casting appropriately.
00186  *
00187  *  With an 8 MHz clock e.g., delays ranging from 125 nanoseconds
00188  *  up to (2**32-1) * 125ns ~= 536,87 seconds are feasible.
00189  *
00190  */
00191 static __inline__ void
00192 _delay_cycles(const double __ticks_d)
00193 {
00194     uint32_t __ticks = (uint32_t)(__ticks_d);
00195     uint32_t __padding;
00196     uint32_t __loops;
00197 
00198     /*
00199      * Special optimization for very
00200      * small delays - not using any register.
00201      */
00202     if( __ticks <= 12 )  {              /* this can be done with 4 opcodes      */
00203         __padding = __ticks;
00204 
00205     /* create a single byte counter */
00206     } else if( __ticks <= 0x400 )  {
00207         __ticks -= 1;                   /* caller needs 1 cycle to init counter */
00208         __loops = __ticks / 4;
00209         __padding = __ticks % 4;
00210         if( __loops != 0 )
00211             _delay_loop_1_x( (uint8_t)__loops );
00212 
00213     /* create a two byte counter */
00214     } else if( __ticks <= 0x40001 )  {
00215         __ticks -= 2;                   /* caller needs 2 cycles to init counter */
00216         __loops = __ticks / 4;
00217         __padding = __ticks % 4;
00218         if( __loops != 0 )
00219             _delay_loop_2_x( (uint16_t)__loops );
00220 
00221     /* create a four byte counter */
00222     } else  {
00223         __ticks -= 4;                   /* caller needs 4 cycles to init counter */
00224         __loops = __ticks / 8;
00225         __padding = __ticks % 8;
00226         if( __loops != 0 )
00227             _delay_loop_3_x( (uint32_t)__loops );
00228     }
00229 
00230     if( __padding ==  1 )  _NOP1();
00231     if( __padding ==  2 )  _NOP2();
00232     if( __padding ==  3 )  _NOP3();
00233     if( __padding ==  4 )  _NOP4();
00234     if( __padding ==  5 )  _NOP5();
00235     if( __padding ==  6 )  _NOP6();
00236     if( __padding ==  7 )  _NOP7();
00237     if( __padding ==  8 )  _NOP8();
00238     if( __padding ==  9 )  _NOP9();
00239     if( __padding == 10 ) _NOP10();
00240     if( __padding == 11 ) _NOP11();
00241     if( __padding == 12 ) _NOP12();
00242 }
